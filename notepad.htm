const editor = document.getElementById('editor'); const lineNumbers = document.getElementById('line-numbers'); let fileHandle = null; // Track the current file for "Save" vs "Save As" function handleInput() { updateLineNumbers(); updateStats(); } function updateLineNumbers() { const lines = editor.value.split('\n').length; let lineStr = ''; for (let i = 1; i <= lines; i++) { lineStr += i + '
'; } lineNumbers.innerHTML = lineStr; } function syncScroll() { lineNumbers.scrollTop = editor.scrollTop; } function updateStats() { const text = editor.value; const cursorPos = editor.selectionStart; const textBefore = text.substring(0, cursorPos); const lines = textBefore.split('\n'); document.getElementById('charCount').innerText = `Chars: ${text.length}`; document.getElementById('lnCol').innerText = `Ln ${lines.length}, Col ${lines[lines.length - 1].length + 1}`; } // --- File System API Implementation --- async function loadFile() { try { // Open file picker [fileHandle] = await window.showOpenFilePicker({ types: [{ description: 'Text Files', accept: { 'text/plain': ['.txt', '.md', '.js', '.html', '.css'] } }], multiple: false }); const file = await fileHandle.getFile(); editor.value = await file.text(); handleInput(); document.title = `${file.name} - Modern Web Notepad`; } catch (err) { console.error("Open cancelled or failed:", err); } } async function saveFile() { try { // If we don't have a handle yet, it's a "Save As" if (!fileHandle) { fileHandle = await window.showSaveFilePicker({ suggestedName: 'note.txt', types: [{ description: 'Text File', accept: { 'text/plain': ['.txt'] } }] }); } // Create a writable stream to the file const writable = await fileHandle.createWritable(); await writable.write(editor.value); await writable.close(); const file = await fileHandle.getFile(); document.title = `${file.name} - Saved`; } catch (err) { console.error("Save cancelled or failed:", err); } } function newFile() { if (confirm("Discard unsaved changes and create a new document?")) { editor.value = ""; fileHandle = null; document.title = "Modern Web Notepad"; handleInput(); } } function findAndReplace() { const f = document.getElementById('findInput').value; const r = document.getElementById('replaceInput').value; if (!f) return; // Using a regex with 'g' flag for global replacement const escapedF = f.replace(/[.*+?^${}()|[\]\s]/g, '\\$&'); editor.value = editor.value.replace(new RegExp(escapedF, 'g'), r); handleInput(); } // Initialize updateLineNumbers();
